## 数据双向绑定示例

```tsx
import React, { useState } from 'react'

function App() {
  const [msg, setMsg] = useState('')

  const inputHandler = (e: React.ChangeEvent<HTMLInputElement>) => {
    setMsg(e.target.value)
  }

  return (
    <div id="App">
        <div>msg: {msg}</div>
        <input value={msg} onInput={inputHandler}/>
    </div>
  )
}

export default App
```
